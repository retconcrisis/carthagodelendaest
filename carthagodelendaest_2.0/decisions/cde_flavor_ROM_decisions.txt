country_decisions = {
	declare_pax_romana = {
		potential = {
			tag = ROM
			NOT = {
				has_variable = pax_romana_declared
			}
		}
		allow = {
			war = no
			has_civil_war = no
			num_of_cities >= 1000
			stability >= 75
		}
		effect = {
			add_country_modifier = {
				name = pax_romana_modifier
			}
			hidden_effect = {
				set_variable = {
					name = pax_romana_declared
				}
				trigger_event = {
					id = "pax_romana.1"
				}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}

	end_pax_romana = {
		potential = {
			tag = ROM
			has_variable = pax_romana_declared
		}
		allow = {
			has_civil_war = no
		}
		effect = {
			add_stability = -20
			remove_country_modifier = pax_romana_modifier
			hidden_effect = {
				remove_variable = {
					name = pax_romana_declared
				}
				trigger_event = {
					id = "pax_romana.2"
				}
			}
		}
	}
	
}
