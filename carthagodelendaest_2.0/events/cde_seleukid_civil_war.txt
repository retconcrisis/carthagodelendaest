			}
		}
		scope:alexanders_army = {
			set_as_commander = char:32

		}

			}

	}

		name = cde_seleukid_civil_war.2.a 	#Support
		
		custom_tooltip = cde_seleukid_civil_war.2.tt

		c:JUD = {
			trigger_event = { 
				id = cde_seleukid_civil_war.6
				days = 15
			}
		}

		c:EGY = {
			trigger_event = { 
				id = cde_seleukid_civil_war.8
			}
		}
	}
