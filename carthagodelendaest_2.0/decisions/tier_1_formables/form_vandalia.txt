			NOT = {

			}
			NOR = {
				is_tier_1_formable_trigger = yes
				is_tier_2_formable_trigger = yes
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {
				OR = {
					AND = {
						has_owner = yes

							primary_culture = vandal
						}
					}
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_vandalia_exists
				NOT = {
					any_country = {
						tag = VAN
					}
				}
			}
			can_form_nation_trigger = yes
			custom_tooltip = {
				text = any_vandal_country_tt

						count = all

							primary_culture = vandal

							NOT = {
								this = root
							}
						}
						is_subject_of = root
					}
					AND = {
						num_of_cities >= 1
						any_country = {

						}
					}
				}
			}
		}

			}

			}
