namespace = cde_flavor_prg

#The Pergamene Donation - for Rome tag
cde_flavor_prg.1 = {
	type = country_event
	title = "cde_flavor_prg.1.t"
	desc = "cde_flavor_prg.1.desc"
	picture = great_work_celebration
	
	fire_only_once = yes
	
	trigger = {
		tag = ROM
		num_of_cities >= 500
		has_civil_war = no
		war = no
		c:PRG = {
			is_ai = yes
			has_civil_war = no
			war = no
			OR = {
				opinion = {
					value >= 100
					target = c:ROM
				}
				current_ruler = {
					is_friend = c:ROM.current_ruler
				}
				current_ruler = {
					is_friend = c:ROM.current_co_ruler
				}
			}
			current_ruler = {
				OR = {
					finesse < 5
					has_trait = submissive
					has_trait = content
				}
				NOT = {
					has_trait = proud
				}
			}
		}
		c:PRG.fam:Attalid = {
			any_family_member = { 
				is_ruler = yes 
				employer = { tag = c:PRG }
			}
		}
		char:78 = {		#Attalos II
            is_alive = no
		}
		OR = {
			char:80 = {		#Attalos III
				is_alive = no
			}
			c:PRG = {
				current_ruler = {	#Attalos III
					this = char:80
				}
			}
		}
	}
	
	option = {
		name = cde_flavor_prg.1.a
		
		custom_tooltip = cde_flavor_prg.1.tt
		
		hidden_effect = {
			c:PRG = {
				#Rome gets Asia region
				every_owned_province = {
					limit = {
						NOR = {
							is_in_region = bithynia_region
							is_in_region = cappadocia_pontica_region
							is_in_region = cappadocia_region
							is_in_region = cilicia_region
							is_in_area = lycaonia_area
							is_in_area = isauria_area
						}
					}
					set_owned_by = c:ROM
				}
				
				#Cappadocia gets Cappadocia Taurica
				every_owned_province = {
					limit = {
						OR = {
							is_in_region = cappadocia_region
							is_in_area = lycaonia_area
							is_in_area = isauria_area
						}
					}
					set_owned_by = c:CAP
				}
				
				#Lycia gets Cilicia
				every_owned_province = {
					limit = {
						OR = {
							is_in_region = cilicia_region
						}
					}
					set_owned_by = c:LYC
				}
				
			}
		}
		
		#Moves Attalid family members to Rome
		c:PRG.fam:Attalid = {
			every_family_member = {
				limit = {
					employer = { tag = PRG }
				}
				move_country = c:ROM
				hidden_effect = {
					remove_character_modifier = foreign_citizen
					set_home_country = c:ROM
				}
			}
		}
		
		set_variable = {
			name = pergamene_donation_completed
		}
	}
}
